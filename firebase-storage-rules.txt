// 临时的 Firebase Storage 安全规则，允许公开访问
// 请在 Firebase 控制台的 Storage > Rules 中使用以下规则：

service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      // 允许所有用户读取和写入（仅用于开发测试）
      allow read, write: if true;
    }
  }
}

// 注意：这个规则允许任何人访问您的 Storage，仅适用于开发环境
// 在生产环境中，请使用更严格的规则，例如：
// allow read, write: if request.auth != null; // 仅允许已认证用户
